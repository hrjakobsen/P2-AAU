\subsection{LSB enhancing}
\begin{frame}{LSB enhancing}
	LSB enhancing
	\begin{itemize}
		\item fjerne alle 7 højtstående bits
		\item Alle bytes har nu værdien 0 eller 1
		\item Alle 0 bytes forbliver 0, imens alle 1 bytes bilver 255
	\end{itemize}
\end{frame}
