\begin{frame}{LSB forstærkning}
	LSB forstærkning
	\begin{itemize}
		\item Fjern de 7 højtstående bits
		\item Alle bytes har nu værdien 0 eller 1
		\item Alle bytes ganges med 255
	\end{itemize}
\end{frame}
