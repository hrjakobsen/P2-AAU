\subsection{LSB enhancing}
\begin{frame}{LSB enhancing}
	LSB enhancing
	\begin{itemize}
		\item eliminate all 7 high-level bits
		\item all bytes are now 0 or 1
		\item 0 stays 0 while 1 becomes 256
	\end{itemize}
